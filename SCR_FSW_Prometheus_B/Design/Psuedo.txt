#include <Lib>

Define global variables

CDH cdh
FlightLogic prometheus

setup()
{
	init CDH, FlightLogic
}

loop()
{
	state = stateCheck()
	switch(state)
	{
		case Standby:
			Standby()
		case Ascent:
			Ascent()
		case Descent:
			Descent()
		case Recovery:
			Recovery()
	}
}

stateCheck()
{
	state check code
}

Standby()
{
	debug, testing, etc
}

Ascent()
{
	cdh.flight()
	prometheus.launchDetect()
	prometheus.apogeeDetect()
}

Descent()
{
	cdh.flight()
	prometheus.mainDetect()
	prometheus.landingDetect()
}

Recovery()
{
	cdh.recovery()
}
